#preprocessing
stop_words = set(stopwords.words('english'))

def clean_text(text):
    text = str(text).lower()
    text = re.sub(r'http\S+|www\S+', '', text)      # remove URLs
    text = re.sub(r'@\w+|#\w+', '', text)            # remove mentions/hashtags
    text = re.sub(r'[^a-z\s]', '', text)             # keep only letters
    text = re.sub(r'\s+', ' ', text).strip()         # clean whitespace
    tokens = word_tokenize(text)
    tokens = [t for t in tokens if t not in stop_words and len(t) > 1]
    return tokens

df['tokens']     = df['text'].apply(clean_text)

#Encoding sequences

vocab      = w2v_model.wv.key_to_index
PAD_IDX    = 0
UNK_IDX    = 1
VOCAB_SIZE = len(vocab) + 2
MAX_LEN    = 100

def tokens_to_ids(tokens):
    ids = [vocab.get(t, UNK_IDX - 1) + 2 for t in tokens]
    if len(ids) >= MAX_LEN:
        return ids[:MAX_LEN]
    return ids + [PAD_IDX] * (MAX_LEN - len(ids))

df['input_ids'] = df['tokens'].apply(tokens_to_ids)

# Build embedding matrix to initialize our neural net weights

embedding_matrix = np.zeros((VOCAB_SIZE, embedding_dim))

class TextCNN(nn.Module):
    def __init__(self, vocab_size, embed_dim, num_filters, 
                 kernel_sizes, num_classes, embedding_matrix=None, dropout=0.5):
        super(TextCNN, self).__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=PAD_IDX)
        if embedding_matrix is not None:
            self.embedding.weight = nn.Parameter(
                torch.tensor(embedding_matrix, dtype=torch.float32)
            )
        self.convs   = nn.ModuleList([
            nn.Conv1d(embed_dim, num_filters, k) for k in kernel_sizes
        ])
        self.dropout = nn.Dropout(dropout)
        self.fc      = nn.Linear(num_filters * len(kernel_sizes), num_classes)

    def forward(self, x):
        x      = self.embedding(x).permute(0, 2, 1)
        pooled = [torch.relu(conv(x)).max(dim=2).values for conv in self.convs]
        x      = torch.cat(pooled, dim=1)
        return self.fc(self.dropout(x))

cnn_model = TextCNN(
    vocab_size=VOCAB_SIZE,
    embed_dim=100,
    num_filters=128,
    kernel_sizes=[2, 3, 4],
    num_classes=2,
    embedding_matrix=embedding_matrix
)

#BiLSTM model
class TextLSTM(nn.Module):
    def __init__(self, vocab_size, embed_dim, hidden_dim, num_layers,
                 num_classes, embedding_matrix=None, dropout=0.5):
        super(TextLSTM, self).__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=PAD_IDX)
        if embedding_matrix is not None:
            self.embedding.weight = nn.Parameter(
                torch.tensor(embedding_matrix, dtype=torch.float32)
            )
        self.lstm    = nn.LSTM(embed_dim, hidden_dim, num_layers=num_layers,
                               batch_first=True, bidirectional=True,
                               dropout=dropout if num_layers > 1 else 0)
        self.dropout = nn.Dropout(dropout)
        self.fc      = nn.Linear(hidden_dim * 2, num_classes)

    def forward(self, x):
        x          = self.embedding(x)
        _, (hn, _) = self.lstm(x)
        hn         = torch.cat([hn[-2], hn[-1]], dim=1)
        return self.fc(self.dropout(hn))

lstm_model = TextLSTM(
    vocab_size=VOCAB_SIZE,
    embed_dim=100,
    hidden_dim=128,
    num_layers=2,
    num_classes=2,
    embedding_matrix=embedding_matrix
)
print(lstm_model)

  (dropout): Dropout(p=0.5, inplace=False)

#Training loop (runs both models)
def train_model(model, train_loader, val_loader, epochs=10, lr=1e-3):
    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
    criterion = nn.CrossEntropyLoss()
    history   = {'train_loss': [], 'val_loss': [], 'val_acc': []}

    for epoch in range(epochs):
        model.train()
        total_loss = 0
        for X_batch, y_batch in train_loader:
            optimizer.zero_grad()
            loss = criterion(model(X_batch), y_batch)
            loss.backward()
            optimizer.step()
            total_loss += loss.item()

        model.eval()
        val_loss, correct = 0, 0
        with torch.no_grad():
            for X_batch, y_batch in val_loader:
                preds     = model(X_batch)
                val_loss += criterion(preds, y_batch).item()
                correct  += (preds.argmax(1) == y_batch).sum().item()

        avg_train = total_loss / len(train_loader)
        avg_val   = val_loss   / len(val_loader)
        val_acc   = correct    / len(val_loader.dataset)

        history['train_loss'].append(avg_train)
        history['val_loss'].append(avg_val)
        history['val_acc'].append(val_acc)
        print(f"Epoch {epoch+1}/{epochs} | Train Loss: {avg_train:.4f} | "
              f"Val Loss: {avg_val:.4f} | Val Acc: {val_acc:.4f}")

    return history

print("=== Training CNN ===")

#Evaluation and confusion matrices
def evaluate_model(model, test_loader, model_name="Model"):
    model.eval()
    all_preds, all_labels = [], []
    with torch.no_grad():
        for X_batch, y_batch in test_loader:
            all_preds.extend(model(X_batch).argmax(1).tolist())
            all_labels.extend(y_batch.tolist())

    print(f"\n=== {model_name} Test Results ===")
    print(classification_report(all_labels, all_preds,
                                 target_names=['No Stress', 'Stress']))

    cm = confusion_matrix(all_labels, all_preds)
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
                xticklabels=['No Stress', 'Stress'],
                yticklabels=['No Stress', 'Stress'])
    plt.title(f'{model_name} Confusion Matrix')
    plt.ylabel('True Label')
    plt.xlabel('Predicted Label')
    plt.tight_layout()
    plt.savefig(f'{model_name.lower().replace(" ","_")}_confusion_matrix.png')
    plt.show()
    return all_preds, all_labels

cnn_preds,  labels = evaluate_model(cnn_model,  test_loader, "CNN")
lstm_preds, _      = evaluate_model(lstm_model, test_loader, "BiLSTM")
